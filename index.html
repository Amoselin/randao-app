// 將 TOTAL 改為 let，初始值可設為預設，稍後會被自動覆蓋 let TOTAL = 15 * 60; const CIRC = 2 * Math.PI * 104; let current = 'home'; let timeLeft = TOTAL; let timer = null; let paused = false; let count = 0; // 獲取音頻元素 const audio = document.getElementById('meditationAudio'); /* ── 自動化時間設定 ── */ // 當音檔資訊載入完成後，自動更新計時器總時間 audio.addEventListener('loadedmetadata', () => { TOTAL = Math.floor(audio.duration); // 自動獲取音檔秒數 (例如 909 秒) timeLeft = TOTAL; updateDisplay(); // 更新介面顯示為 15:09 updateRing(); // 重置進度環 }); /* ── Storage ── */ function loadCount() { try { count = parseInt(localStorage.getItem('randao_count') || '0', 10) || 0; } catch(e) { count = 0; } const el = document.getElementById('countLabel'); el.textContent = count > 0 ? `您已在此燃道修行了 ${count} 次` : '開始您的第一次修行'; } function saveCount() { try { localStorage.setItem('randao_count', String(count)); } catch(e) {} } /* ── Navigation ── */ function goTo(page) { const from = document.getElementById(current); const to = document.getElementById(page); from.classList.remove('visible'); from.classList.add('hidden'); requestAnimationFrame(() => { to.classList.remove('hidden'); requestAnimationFrame(() => to.classList.add('visible')); }); current = page; document.getElementById('rippleWrap').style.display = page === 'meditation' ? 'block' : 'none'; } /* ── Meditation ── */ function startMeditation() { clearInterval(timer); // 每次開始都以音檔最新長度為準 if (!isNaN(audio.duration)) { TOTAL = Math.floor(audio.duration); } timeLeft = TOTAL; paused = false; document.getElementById('pauseBtn').textContent = '暫停冥想'; // 播放音頻 audio.currentTime = 0; audio.play().catch(e => console.error("播放受阻:", e)); updateDisplay(); updateRing(); goTo('meditation'); timer = setInterval(tick, 1000); } function tick() { if (paused) return; timeLeft = Math.max(0, timeLeft - 1); updateDisplay(); updateRing(); if (timeLeft <= 0) { clearInterval(timer); audio.pause(); audio.currentTime = 0; count++; saveCount(); setTimeout(() => { loadCount(); const c = document.getElementById('completion'); const clone = c.cloneNode(true); c.parentNode.replaceChild(clone, c); goTo('completion'); }, 500); } } function updateDisplay() { const m = String(Math.floor(timeLeft / 60)).padStart(2, '0'); const s = String(timeLeft % 60).padStart(2, '0'); document.getElementById('timerDisplay').textContent = m + ':' + s; } function updateRing() { const pct = timeLeft / TOTAL; const offset = CIRC * (1 - pct); document.getElementById('progressRing') .setAttribute('stroke-dashoffset', offset.toFixed(2)); } function togglePause() { paused = !paused; document.getElementById('pauseBtn').textContent = paused ? '繼續冥想' : '暫停冥想'; if (paused) { audio.pause(); } else { audio.play(); } } function showStopConfirm() { document.getElementById('stopOverlay').classList.remove('hidden'); } function hideStopConfirm() { document.getElementById('stopOverlay').classList.add('hidden'); } function confirmStop() { clearInterval(timer); hideStopConfirm(); audio.pause(); audio.currentTime = 0; timeLeft = TOTAL; paused = false; document.getElementById('pauseBtn').textContent = '暫停冥想'; updateDisplay(); updateRing(); goTo('selection'); } function goHome() { clearInterval(timer); audio.pause(); audio.currentTime = 0; goTo('home'); } loadCount();
